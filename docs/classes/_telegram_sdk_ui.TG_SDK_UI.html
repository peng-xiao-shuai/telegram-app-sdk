    • options: OpenPayPopupPayload
    • payTypes: {
          amount: string;

      }[]

      支付类型

    Returns void

    Remarks

    唤起支付弹窗 (兼容浏览器以及TG)。非 TG 使用 Stars 支付会直接返回成功

    Example

    window.TG_SDK.openPayPopup({
       message: '首冲礼包 ￥6 获得 xxx 钻石',
       amount: '1',
       order_id: String(Math.round(Math.random() * 10000)),
       start: () => {
         console.log('开始支付');
       },
       result: (status) => {
         console.log('支付状态 status', status);
       },
     },
     [
       {
         token: 'Ton',
         amount: '0.01',
       },
       {
         token: 'Stars',
         amount: '1',
       },
     ])
